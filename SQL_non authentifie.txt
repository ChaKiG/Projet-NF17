//Rubrique permettant d’accéder aux compétitions
//d’un jour donné par l’intermédiaire d’un calendrier

SELECT date, nom
FROM competition
GROUP BY date 
ORDER BY date;

SELECT nom, date
FROM competition
GROUP BY nom 
ORDER BY nom;

SELECT date, nom 
FROM competition
WHERE competition.nom = $nom 
ORDER BY date;

FROM competition

//Sélection d’une compétition pour accéder aux
//statistiques de cette compétition
// <=> Obtenir la liste de toutes confrontations et résultats (classement par karatéka et par club) d'une compétition en cours ou terminée

SELECT gagnant, perdant, scoreGagnant, scorePerdant, competition.clubNom
FROM confrontation, competition
WHERE confrontation.nomCompetition = competition.nom 
AND confrontation.nomCompetition = $nom
GROUP BY gagnant, competition.clubNom ;
